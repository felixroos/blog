# Graphs

## Graphviz

Graphviz json renderer Using [@hpcc-js/wasm](https://www.npmjs.com/package/@hpcc-js/wasm) + [jgf-dot](https://github.com/jsongraph/jgf-dot#jgf-dot).

More info, see [Graphviz](../components/Graphviz).

import { Graph } from '../../components/graphs/Graphviz.tsx';

<Graph
  editable={false}
  json={{
    graph: {
      directed: true,
      nodes: [
        {
          label: 'A',
          fillcolor: 'steelblue',
          style: 'filled',
          id: 0,
        },
        {
          label: 'B',
          fillcolor: 'salmon',
          style: 'filled',
          id: 1,
        },
      ],
      edges: [
        {
          source: 0,
          target: 1,
          label: 'link',
        },
      ],
    },
  }}
/>
